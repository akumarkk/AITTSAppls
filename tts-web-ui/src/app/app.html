// src/app/services/tts.service.ts import { Injectable, signal } from '@angular/core'; import { HttpClient } from '@angular/common/http'; import { firstValueFrom } from 'rxjs'; @Injectable({ providedIn: 'root' }) export class TtsService { private readonly backendUrl = 'http://localhost:5005/v1/audio/speech'; // Track state for the UI isProcessing = signal(false); private currentAudio: HTMLAudioElement | null = null; private currentObjectURL: string | null = null; constructor(private http: HttpClient) {} async speak(text: string, voice: string = 'tara'): Promise { this.stop(); // Stop any currently playing audio this.isProcessing.set(true); try { // 1. Request audio blob from local Orpheus server const blob = await firstValueFrom( this.http.post(this.backendUrl, { input: text, voice: voice, model: "orpheus-3b", response_format: "wav" }, { responseType: 'blob' }) ); // 2. Create a URL for the blob this.currentObjectURL = URL.createObjectURL(blob); this.currentAudio = new Audio(this.currentObjectURL); // 3. Play and wait for it to finish await this.currentAudio.play(); this.currentAudio.onended = () => this.cleanup(); } catch (error) { console.error('TTS Generation failed:', error); } finally { this.isProcessing.set(false); } } stop(): void { if (this.currentAudio) { this.currentAudio.pause(); this.currentAudio = null; } this.cleanup(); } private cleanup(): void { if (this.currentObjectURL) { URL.revokeObjectURL(this.currentObjectURL); this.currentObjectURL = null; } } }
